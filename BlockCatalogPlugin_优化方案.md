// 点击"拾取"按钮后
private void BtnPickBlock_Click(object sender, EventArgs e)
{
    var ed = Application.DocumentManager.MdiActiveDocument.Editor;
    var opt = new PromptEntityOptions("\n选择图框块: ");
    opt.SetRejectMessage("\n必须选择块参照");
    opt.AddAllowedClass(typeof(BlockReference), true);

    var result = ed.GetEntity(opt);
    if (result.Status == PromptStatus.OK)
    {
        using var tr = ed.Document.Database.TransactionManager.StartTransaction();
        var br = tr.GetObject(result.ObjectId, OpenMode.ForRead) as BlockReference;
        if (br != null)
        {
            cmbFilterBlockName.SelectedItem = br.Name;
            AppendLog($"已拾取块: {br.Name}");
        }
    }
}

**鼠标拖拽定义尺寸功能**：
```csharp
// 在CAD中拖拽鼠标定义表格尺寸
public class MouseSizeDefinition
{
    public double Width;   // 用户拖拽定义的宽度
    public double Height;  // 用户拖拽定义的高度
    public Point3d Corner; // 右下角点

    public static MouseSizeDefinition GetFromMouse(Editor ed)
    {
        // 1. 提示用户点击左上角
        var startResult = ed.GetPoint("\n指定表格左上角: ");
        if (startResult.Status != PromptStatus.OK) return null;

        // 2. 动态拖拽到右下角
        var dragOpt = new PromptCornerOptions("\n拖拽到右下角: ", startResult.Value);
        var dragResult = ed.GetCorner(dragOpt);
        if (dragResult.Status != PromptStatus.OK) return null;

        // 3. 计算尺寸
        var start = startResult.Value;
        var end = dragResult.Value;
        return new MouseSizeDefinition
        {
            Width = Math.Abs(end.X - start.X),
            Height = Math.Abs(end.Y - start.Y),
            Corner = end
        };
    }
}
```

private void AutoAdjustColumnWidths()
{
    // 1. 遍历每列计算内容最大宽度
    foreach (DataGridViewColumn col in dgvPreview.Columns)
    {
        int maxWidth = col.HeaderText.Length * 10; // 基于标题
        foreach (DataGridViewRow row in dgvPreview.Rows)
        {
            if (row.Cells[col.Index].Value != null)
            {
                var cellText = row.Cells[col.Index].Value.ToString();
                int textWidth = TextRenderer.MeasureText(cellText, dgvPreview.Font).Width + 10;
                maxWidth = Math.Max(maxWidth, textWidth);
            }
        }
        col.Width = Math.Min(maxWidth, 300); // 限制最大300px
    }

    // 2. 计算总宽度和比例
    int totalWidth = dgvPreview.Columns.GetColumnsWidth(DataGridViewElementStates.Visible);
    lblInfo.Text = $"共 {_blocks.Count} 行 | 预览宽度: {totalWidth}px";
}

**表达式编辑器功能**：
```
支持的变量和函数：
{n}     - 纯数字序号 (1, 2, 3...)
{nn}    - 带前导零序号 (01, 02, 03...)
{值}    - 当前属性原值
{图名}  - 读取图名属性值
{布局}  - 布局名称
{块名}  - 块定义名称

类型缩写映射：
平面图 → PL
立面图 → LM
剖面图 → PM
详图   → XT

示例表达式：
{前缀}-{nn}-{类型缩写(图名)}  → 建施-01-PL
{布局}-{n}                    → Layout1-1
```

## 技术要点

### 生成后指定位置
```csharp
private void GenerateCatalog()
{
    // 1. 隐藏面板
    Plugin.HidePalette();

    // 2. 提示用户选择位置
    var ed = Application.DocumentManager.MdiActiveDocument.Editor;
    var pointOpt = new PromptPointOptions("\n指定目录插入位置: ");
    var pointResult = ed.GetPoint(pointOpt);

    if (pointResult.Status == PromptStatus.OK)
    {
        // 3. 在指定位置生成
        _generator.GenerateTable(_currentResult.Blocks, _currentStyle,
            null, pointResult.Value);
        AppendLog($"目录已生成到 ({pointResult.Value.X:F1}, {pointResult.Value.Y:F1})");
    }

    // 4. 显示面板
    Plugin.ShowPalette();
}
```
